#include <stdio.h>
int main() {
    int i,x,n;
    int count=0;
    scanf("%d",&n);
    int marks[n];
    for(i=0;i<n;i++){
        scanf("%d",&marks[i]);
    }
    scanf("%d",&x);
    for(i=0;i<n;i++){
        if(marks[i]==x){
            count++;
        }
    }
    printf("Frequency count:%d\n",count);
    return 0;
}
