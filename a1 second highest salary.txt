#include <stdio.h>
#include <limits.h>
int main() {
    int i,j,n;
    scanf("%d",&n);
    int salary[n];
    for (int i=0;i<n;i++){
        scanf("%d",&salary[i]);
    }
    int l=INT_MIN;
    int sl=INT_MIN;
    for(int i=0;i<n;i++){
        if(salary[i]>l){
            sl=l;
            l=salary[i];
        }
        else if(salary[i]>sl && salary[i]!=l){
            sl=salary[i];
        }
    }
    if(sl==INT_MIN){
        printf("No second highest salary found.\n");
    }
    else{

    }

}
